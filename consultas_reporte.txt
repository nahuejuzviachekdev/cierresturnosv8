17) Diferencias      (CORREGIR)
WITH diferencia_teorica AS (
    SELECT diferencia
    FROM cierre_turno_resumen_diferencias
    WHERE id_cierre_turno = 424510
),
factura_pendiente AS (
    SELECT SUM(diferencia * precio) AS importe
    FROM cierre_turno_aforadores_resumen
    WHERE id_cierre_turno = 424510
)
SELECT
    'Diferencia teórica:' AS "Concepto",
    dt.diferencia AS "Importe"
FROM diferencia_teorica dt
UNION ALL
SELECT
    'Factura pendiente:',
    fp.importe
FROM factura_pendiente fp
UNION ALL
SELECT
    'Diferencia real:',
    fp.importe - dt.diferencia
FROM diferencia_teorica dt, factura_pendiente fp;
